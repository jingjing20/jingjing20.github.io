function unique(arr) {
	var fomArr = arr.sort()
	var newArr = []
	for (let i = 0; i < arr.length; i++) {
		if (fomArr[i] !== fomArr[i - 1]) {
			newArr.push(fomArr[i])
		}
	}
	return newArr
}
